## 六、构建 CNN 模型

### 模型架构

```python
def create_model(input_shape=(28, 28, 1), num_classes=10):
    """创建 CNN 模型"""
    
    inputs = keras.Input(shape=input_shape)
    
    # 数据增强
    x = data_augmentation(inputs)
    
    # 卷积块 1
    x = layers.Conv2D(32, (3, 3), activation='relu', padding='same')(x)
    x = layers.Conv2D(32, (3, 3), activation='relu', padding='same')(x)
    x = layers.MaxPooling2D((2, 2))(x)
    x = layers.Dropout(0.25)(x)
    
    # 卷积块 2
    x = layers.Conv2D(64, (3, 3), activation='relu', padding='same')(x)
    x = layers.Conv2D(64, (3, 3), activation='relu', padding='same')(x)
    x = layers.MaxPooling2D((2, 2))(x)
    x = layers.Dropout(0.25)(x)
    
    # 卷积块 3
    x = layers.Conv2D(128, (3, 3), activation='relu', padding='same')(x)
    x = layers.Conv2D(128, (3, 3), activation='relu', padding='same')(x)
    x = layers.MaxPooling2D((2, 2))(x)
    x = layers.Dropout(0.25)(x)
    
    # 全连接层
    x = layers.Flatten()(x)
    x = layers.Dense(256, activation='relu')(x)
    x = layers.BatchNormalization()(x)
    x = layers.Dropout(0.5)(x)
    
    # 输出层
    outputs = layers.Dense(num_classes, activation='softmax')(x)
    
    model = models.Model(inputs, outputs, name='fashion_cnn')
    
    return model

# 创建模型
model = create_model()
model.summary()
```

## 七、编译和训练

### 回调函数

```python
# 学习率调度
def lr_schedule(epoch, lr):
    if epoch < 10:
        return lr
    elif epoch < 20:
        return lr * 0.5
    else:
        return lr * 0.1

lr_scheduler = keras.callbacks.LearningRateScheduler(lr_schedule)

# 早停
early_stopping = keras.callbacks.EarlyStopping(
    monitor='val_loss',
    patience=10,
    restore_best_weights=True,
    verbose=1
)

# 模型检查点
checkpoint = keras.callbacks.ModelCheckpoint(
    'best_model.h5',
    monitor='val_accuracy',
    save_best_only=True,
    mode='max',
    verbose=1
)

# TensorBoard
tensorboard = keras.callbacks.TensorBoard(
    log_dir='./logs',
    histogram_freq=1
)
```

### 编译模型

### 加载模型进行预测

```python
# 加载模型
loaded_model = keras.models.load_model('fashion_mnist_cnn.h5')

# 预测函数
def predict_image(image):
    """
    预测单张图像
    
    Args:
        image: 28x28 灰度图像 (numpy array)
    
    Returns:
        预测类别名称和置信度
    """
    # 预处理
    img = image.astype('float32') / 255.0
    img = np.expand_dims(img, axis=-1)
    img = np.expand_dims(img, axis=0)
    
    # 预测
    pred_proba = loaded_model.predict(img, verbose=0)[0]
    pred_class = np.argmax(pred_proba)
    confidence = pred_proba[pred_class]
    
    return class_names[pred_class], confidence

# 测试
test_idx = 0
test_image = x_test[test_idx].squeeze()
pred_name, conf = predict_image(test_image)
print(f"预测：{pred_name} (置信度：{conf:.2%})")
print(f"真实：{class_names[y_true[test_idx]]}")
```

from io import BytesIO
from PIL import Image

app = Flask(__name__)

@app.route('/predict', methods=['POST'])
def predict():
    # 获取图像
    data = request.json
    image_data = base64.b64decode(data['image'])
    image = Image.open(BytesIO(image_data)).convert('L')
    image = image.resize((28, 28))
    image_array = np.array(image)
    
    # 预测
    pred_name, confidence = predict_image(image_array)
    
    return jsonify({
        'prediction': pred_name,
        'confidence': float(confidence)
    })

if __name__ == '__main__':
